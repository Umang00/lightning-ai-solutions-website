```python
def calculate_personality_scores(birth_chart):
    scores = {}
    
    # Emotional Depth Score
    emotional_depth = 0
    if birth_chart.moon_sign in ['Cancer', 'Scorpio', 'Pisces']:
        emotional_depth += 30
    
    water_planets = count_planets_in_element(birth_chart, 'Water')
    emotional_depth += water_planets * 10
    
    if has_strong_4th_house(birth_chart):
        emotional_depth += 20
    
    if moon_pluto_aspect(birth_chart):
        emotional_depth += 15
    
    scores['emotional_depth'] = min(100, emotional_depth)
    
    # Social Adaptability
    social_score = 0
    air_planets = count_planets_in_element(birth_chart, 'Air')
    social_score += air_planets * 8
    
    if has_strong_11th_house(birth_chart):
        social_score += 20
    
    venus_score = get_venus_score(birth_chart)
    social_score += venus_score
    
    scores['social_adaptability'] = min(100, social_score)
    
    # Repeat for all personality dimensions...
    
    # Generate natural language descriptions
    descriptions = {}
    for metric, score in scores.items():
        descriptions[metric] = generate_description(metric, score, birth_chart)
    
    return {
        'scores': scores,
        'descriptions': descriptions,
        'comparisons': get_population_percentiles(scores)
    }
```

```python
def calculate_birth_chart(birth_date, birth_time, lat, lon, tz):
    """
    Calculate complete Vedic birth chart
    """
    # Convert birth datetime to UTC Julian Day
    dt = datetime.combine(birth_date, birth_time)
    dt_tz = timezone(tz).localize(dt)
    dt_utc = dt_tz.astimezone(timezone('UTC'))
    
    jd = swe.julday(
        dt_utc.year, dt_utc.month, dt_utc.day,
        dt_utc.hour + dt_utc.minute/60.0
    )
    
    # Calculate Lahiri Ayanamsa (sidereal correction)
    swe.set_sid_mode(swe.SIDM_LAHIRI)
    ayanamsa = swe.get_ayanamsa(jd)
    
    # Calculate house cusps (Whole Sign system)
    houses, ascendant = swe.houses(jd, lat, lon, b'W')  # W = Whole Sign
    
    # Calculate planetary positions
    planets = {}
    planet_list = [
        (swe.SUN, 'Sun'),
        (swe.MOON, 'Moon'),
        (swe.MARS, 'Mars'),
        (swe.MERCURY, 'Mercury'),
        (swe.JUPITER, 'Jupiter'),
        (swe.VENUS, 'Venus'),
        (swe.SATURN, 'Saturn'),
        (swe.TRUE_NODE, 'Rahu'),  # North Node
        (swe.MEAN_NODE + 180, 'Ketu')  # South Node
    ]
    
    for planet_id, planet_name in planet_list:
        # Get tropical position
        result = swe.calc_ut(jd, planet_id)
        tropical_lon = result[0][0]
        
        # Convert to sidereal (Vedic)
        sidereal_lon = (tropical_lon - ayanamsa) % 360
        
        # Determine sign and house
        sign = get_sign_from_longitude(sidereal_lon)
        house = get_house_from_longitude(sidereal_lon, ascendant)
        
        # Determine nakshatra (for Moon)

        }

        'ascendant': get_sign_from_longitude((ascendant[0] - ayanamsa) % 360),
        'ascendant_degree': (ascendant[0] - ayanamsa) % 360,

        'd9_navamsa': d9_navamsa,

def get_sign_from_longitude(lon):
    """Convert longitude to zodiac sign"""
    signs = ['Aries', 'Taurus', 'Gemini', 'Cancer', 'Leo', 'Virgo',
             'Libra', 'Scorpio', 'Sagittarius', 'Capricorn', 'Aquarius', 'Pisces']
    return signs[int(lon / 30)]

def get_nakshatra(lon):
    """Get nakshatra from longitude (27 nakshatras, each 13¬∞20')"""
    nakshatras = [
        'Ashwini', 'Bharani', 'Krittika', 'Rohini', 'Mrigashira', 'Ardra',
        'Punarvasu', 'Pushya', 'Ashlesha', 'Magha', 'Purva Phalguni', 'Uttara Phalguni',
        'Hasta', 'Chitra', 'Swati', 'Vishakha', 'Anuradha', 'Jyeshtha',
        'Mula', 'Purva Ashadha', 'Uttara Ashadha', 'Shravana', 'Dhanishta', 'Shatabhisha',
        'Purva Bhadrapada', 'Uttara Bhadrapada', 'Revati'
    ]
    nakshatra_index = int(lon / (360 / 27))
    return nakshatras[nakshatra_index]

def calculate_aspects(planets):
    """Calculate planetary aspects (Vedic aspects differ from Western)"""
    aspects = []
    
    # Vedic aspects: conjunction (0¬∞), opposition (180¬∞), trine (120¬∞), square (90¬∞)
    # Plus special aspects: 7th aspect (all planets), 4th & 8th (Mars, Saturn, Jupiter)
    
    for p1_name, p1_data in planets.items():
        for p2_name, p2_data in planets.items():
            if p1_name >= p2_name:  # Avoid duplicates
                continue
            
            angle = abs(p1_data['longitude'] - p2_data['longitude'])
            if angle > 180:
                angle = 360 - angle
            
            aspect_type = None
            if angle < 10:  # Conjunction (0¬∞ ¬± 10¬∞)
                aspect_type = 'Conjunction'
            elif 170 < angle < 190:  # Opposition (180¬∞ ¬± 10¬∞)
                aspect_type = 'Opposition'
            elif 110 < angle < 130:  # Trine (120¬∞ ¬± 10¬∞)
                aspect_type = 'Trine'
            elif 80 < angle < 100:  # Square (90¬∞ ¬± 10¬∞)
                aspect_type = 'Square'
            
            if aspect_type:
                aspects.append({
                    'planet1': p1_name,
                    'planet2': p2_name,
                    'type': aspect_type,
                    'angle': angle,
                    'strength': calculate_aspect_strength(angle, aspect_type)
                })
    
    return aspects

def calculate_navamsa(planets):
    """Calculate D-9 (Navamsa) divisional chart for marriage/relationships"""
    navamsa = {}
    for planet_name, planet_data in planets.items():
        lon = planet_data['longitude']
        # Navamsa divides each sign into 9 parts (3¬∞20' each)
        navamsa_lon = (lon * 9) % 360
        navamsa[planet_name] = {
            'sign': get_sign_from_longitude(navamsa_lon),
            'longitude': navamsa_lon
        }
    return navamsa
```

```python
        if not self.parse_consent(consent):
            return await self.end_call_politely()
        
        # Main interview loop
        for question in self.template.questions:
            await self.ask_question(question)
            response = await self.listen_and_transcribe()
            self.conversation_history.append({"q": question, "a": response})
            
            # Generate and ask follow-up if needed
            if self.needs_followup(response):
                followup = await self.generate_followup(question, response)
                await self.ask_question(followup)
                followup_response = await self.listen_and_transcribe()
                self.conversation_history.append({
                    "q": followup, 
                    "a": followup_response
                })
        
        # Thank and end
        await self.speak("Thank you so much for your time! Your feedback 
                         helps us improve Hunch for everyone.")
        await self.end_call()
        
        # Post-processing
        await self.save_transcript()
        await self.analyze_sentiment()
        await self.extract_insights()
```

```python
class ConversationStarterGenerator:
    def __init__(self):
        self.model = "ft:gpt-3.5-turbo:hunch:conversation-coach"
        self.client = OpenAI()
        self.cache = RedisCache()
    
    def generate_starters(self, user_mbti, match_mbti, 
                         compatibility_score, profile_data):
        # Check cache first
        cache_key = f"{user_mbti}:{match_mbti}:{compatibility_score}"
        if cached := self.cache.get(cache_key):
            return cached
        
        # Generate using fine-tuned model
        prompt = self._build_prompt(
            user_mbti, match_mbti, 
            compatibility_score, profile_data
        )
        
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": self._system_prompt()},
                {"role": "user", "content": prompt}
            ],
            temperature=0.8,  # Creativity
            max_tokens=200,
            n=3  # Generate 3 options
        )
        
        starters = [choice.message.content for choice in response.choices]
        
        # Cache for 24 hours
        self.cache.set(cache_key, starters, ttl=86400)
        
        return starters
    
    def _system_prompt(self):
        return """You are an expert dating conversation coach who understands 
        MBTI personality dynamics. Generate natural, engaging conversation 
        starters that:
        1. Reference specific profile details
        2. Match the user's communication style
        3. Appeal to the match's personality type
        4. Encourage meaningful dialogue
        5. Feel authentic, not scripted
        """
```
